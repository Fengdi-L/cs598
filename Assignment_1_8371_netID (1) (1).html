This assignment is related to the simulation study described in Section 2.3.1 (the so-called Scenario 2 or Example 2) of Elements of Statistical Learning (ESL).

Scenario 2: the two-dimensional data $X \in R^2$ in each class are generated from a mixture of 10 different bivariate Gaussian distributions with uncorrelated components and different means, i.e., $$ X | Y = k, Z = j \sim N(m_{kj}, s^2I_2) $$, where $k=0$ or $1$, and $j = 1, 2, ..., 10$. Set $$ P(Y = k) = 1/2, P(Z = j) = 1/10, s^2 = 1/5 $$. In other words, given $Y = k, X$ follows a mixture distribution with probability density function (PDF), $$ \frac{1}{10} \sum_{j=1}^{10} (\frac{1}{\sqrt{2 \pi s^2}})^2 e^{-\frac{\parallel x - m_{kj}\parallel^2}{2 s^2}} $$

  1. Question: Explain how you handle distance ties and voting ties
  • distance ties may occur when you have multiple (training) observations that are equidistant from a test observation.
  • voting ties may occur when K is an even number and you have 50% of the k-nearest-neighbors from each of the two classes.

Answer

  • distance ties: Distance ties occur when multiple training samples are equal distance from a test sample. User might end up with more than K nearest neighbors. I am using random selection to handle distance ties. If multiple points are equal distance, I just randomly choose from the tied points to maintain the exact K number of nearest neighbors. This is my approach for handling distance ties.

  • voting ties: If users are using KNN with an even value of K, a tie can occur when an equal number of nearest neighbors belong to each class. In the case of a tie in K-Nearest Neighbors (KNN), if you're using scipy.stats.mode(), it will return the smallest of the tied values. So, in a binary classification setting where the classes are labeled as 0 and 1, if there's a tie between the two classes (an equal number of neighbors from each class), class 0 will be returned by default.

# Function to compute the Euclidean distance matrix without loops
def compute_distance_matrix(X, Z):
    u = np.sum(X**2, axis=1)  # Squared norms of each row in X, shape (m,)
    v = np.sum(Z**2, axis=1)  # Squared norms of each row in Z, shape (n,)
    C = np.dot(X, Z.T)        # Dot product matrix between X and Z, shape (m, n)
    
    # The distance matrix: (m, 1) + (n,) - 2 * (m, n), shape (m, n)
    D = np.sqrt(u[:, np.newaxis] - 2 * C + v)  
    return D

# k-Nearest Neighbors (kNN) function
def k_nearest_neighbors(train_samples, train_labels, test_samples, k):
    # Step 1: Compute the Euclidean distance matrix between test and training samples
    D = compute_distance_matrix(test_samples, train_samples)
    
    # Step 2: Get the indices of the k-nearest neighbors for each test point
    nearest_neighbors_indices = np.argsort(D, axis=1)[:, :k]
    
    # Step 3: Get the labels of the nearest neighbors
    nearest_neighbors_labels = train_labels[nearest_neighbors_indices]
    
    # Step 4: Voting (mode) to determine the predicted label for each test point
    predictions, _ = mode(nearest_neighbors_labels, axis=1, keepdims=False)  # Set keepdims=False
    
    return predictions.flatten()

# Test the kNN function with k = 1, 3, 5 and compare the results with sklearn's knn
for k in [1, 3, 5]:
    predictions = k_nearest_neighbors(train_samples, train_labels, test_samples, k)
    cm = confusion_matrix(test_labels, predictions)
    test_error = 1 - np.mean(predictions == test_labels)  # Error = 1 - Accuracy
    print(f"Confusion Matrix for k={k}:\n", cm)
    print(f"Test Error for k={k}: {test_error}")

import numpy as np
from sklearn.model_selection import KFold
from sklearn.metrics import confusion_matrix
from scipy.stats import mode  # Import mode from scipy.stats
from scipy.spatial import distance

# kNN prediction function with distance matrix computation
def knn_predict(X_train, y_train, X_test, K):
    # Compute Euclidean distance matrix between X_test and X_train
    dists = distance.cdist(X_test, X_train, 'euclidean')
    
    # Get the indices of the K closest neighbors for each test point
    neighbor_indices = np.argsort(dists, axis=1)[:, :K]
    
    # Get the labels of the K nearest neighbors
    neighbor_labels = y_train[neighbor_indices]
    
    # Predict the most common class (using scipy.stats.mode)
    predictions, _ = mode(neighbor_labels, axis=1, keepdims=False)  # Set keepdims=False
    
    return predictions.flatten()

# Perform 10-fold cross-validation to select the best K
def cross_validate_knn(X, y, candidate_K_values):
    kf = KFold(n_splits=10, shuffle=True, random_state=8371)
    avg_errors = np.zeros(len(candidate_K_values))
    
    # Loop over K values
    for i, K in enumerate(candidate_K_values):
        fold_errors = []
        
        # Perform cross-validation
        for train_index, val_index in kf.split(X):
            X_train, X_val = X[train_index], X[val_index]
            y_train, y_val = y[train_index], y[val_index]
            
            # Get predictions for the validation set
            y_pred = knn_predict(X_train, y_train, X_val, K)
            
            # Compute error rate
            fold_error = np.mean(y_pred != y_val)
            fold_errors.append(fold_error)
        
        # Average the error rates for this K
        avg_errors[i] = np.mean(fold_errors)
    
    return avg_errors

# Generate the candidate K values from 1 to 180
candidate_K_values = list(range(1, 181))

# Run cross-validation to find the best K
cv_errors = cross_validate_knn(train_samples, train_labels, candidate_K_values)

# Find the minimum CV error and the largest K that gives the minimum error
min_cv_error = np.min(cv_errors)
best_K_candidates = [K for i, K in enumerate(candidate_K_values) if cv_errors[i] == min_cv_error]
best_K = max(best_K_candidates)

print(f"Best K value selected: {best_K}")

# Apply KNN with the best K on the test data
y_test_pred = knn_predict(train_samples, train_labels, test_samples, best_K)

# Report results using a confusion matrix
conf_matrix = confusion_matrix(test_labels, y_test_pred)
print("Confusion Matrix (2x2 table):\n", conf_matrix)

# Compute the test error
test_error = np.mean(y_test_pred != test_labels)
print(f"Test Error: {test_error:.4f}")

Part 4: Bayes rule

  1. Implement the Bayes rule. Your implementation should meet the following requirements:
    • Do not use explicit loops over the test sample size (10,000 or 5,000).
    • You are allowed to use loops over the number of centers (10 or 20), although you can avoid all loops.
  2. Test your code with the test data you just generated. (Note that you do not need training data for the Bayes rule.) Report your results (on the test data) as a 2-by-2 table.

The Bayes rule for binary classification (under the zero-one loss), as derived in class, is: predict $Y$ to be 1, if

$$ P(Y = 1 \mid X = x) \ge P(Y = 0 \mid X=x), $$

or equivalently

$$ \frac{P(Y = 1 \mid X = x)}{P(Y = 0 \mid X=x)} \ge 1.$$

Following the data generation process, we have $$ \displaystyle \frac{P(Y=1\mid X=x)}{P(Y=0\mid X=x)}=\frac{P(Y=1) \cdot P(X=x\mid Y=1)}{P(Y=0) \cdot P(X=x\mid Y=0)} $$ $$\displaystyle =\frac{(1/2)\cdot 10^{-1}\sum_{l=1}^{10}(2\pi s^2)^{-1}\exp\left(-\lVert\mathbf{x}-\mathbf{m}_{1l}\rVert^2/(2s^2)\right)}{(1/2)\cdot 10^{-1}\sum_{l=1}^{10}(2\pi s^2)^{-1}\exp\left(-\lVert\mathbf{x}-\mathbf{m}_{0l}\rVert^2/(2s^2)\right)} $$ $$\displaystyle =\frac{\sum_{l=1}^{10}\exp\left(-\lVert\mathbf{x}-\mathbf{m}_{1l}\rVert^2/(2s^2)\right)}{\sum_{l=1}^{10}\exp\left(-\lVert\mathbf{x}-\mathbf{m}_{0l}\rVert^2/(2s^2)\right)}. $$

# Define the likelihood computation (or log-likelihood for numerical stability)
def calculate_log_prob(data, mu):
    # Compute the squared Euclidean distance between data and class centers
    result = np.linalg.norm(data[:, None] - mu, axis=2) ** 2
    # Log-likelihood: - (distance^2) / (2 * variance), using log-sum-exp for numerical stability
    result = -result / (2 * s ** 2)
    return np.log(np.sum(np.exp(result), axis=1))  # Log-sum-exp

def bayes_rule(data, mu0, mu1):
    # Apply Bayes rule based on the log-probabilities for numerical stability
    return np.where(calculate_log_prob(data, mu1) >= calculate_log_prob(data, mu0), 1, 0)

# Make predictions using the Bayes rule
y_pred_bayes = bayes_rule(test_samples, class_1_centers, class_2_centers)

# Print the confusion matrix using sklearn's confusion_matrix

# Test error calculation
test_error = np.mean(y_pred_bayes != test_labels)
print(f"Test Error: {test_error:.4f}")

def knn_fixed_k(train_samples, train_labels, test_samples, k=7):

# Procedure 2: kNN with K chosen by 10-fold CV (from Part 3)
def knn_cv(train_samples, train_labels, test_samples):
    candidate_K_values = list(range(1, 181))
    kf = KFold(n_splits=10, shuffle=True, random_state=8371)
    avg_errors = np.zeros(len(candidate_K_values))
    
    for i, K in enumerate(candidate_K_values):
        fold_errors = []
        for train_idx, val_idx in kf.split(train_samples):
            X_train, X_val = train_samples[train_idx], train_samples[val_idx]
            y_train, y_val = train_labels[train_idx], train_labels[val_idx]
            
            knn = KNeighborsClassifier(n_neighbors=K)
            knn.fit(X_train, y_train)
            y_pred_val = knn.predict(X_val)
            fold_error = np.mean(y_pred_val != y_val)
            fold_errors.append(fold_error)
        
        avg_errors[i] = np.mean(fold_errors)

    knn.fit(train_samples, train_labels)
